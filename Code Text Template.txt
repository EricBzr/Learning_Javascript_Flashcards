<!-- ----------------------------------> || Variables ||
<!-- ---------------------------------->
<!-- ----------------------------------> // Declaring variables
<!-- ----------------------------------> let currentTemp = 22; (let variables are blocked scoped)
<!-- ----------------------------------> var myAge = 23;       (var variables are globally scoped)
<!-- ---------------------------------->
<!-- ----------------------------------> //Declaring a variable without initializing a value
<!-- ----------------------------------> let favNumber; (favNumber = undefined)
<!-- ---------------------------------->
<!-- ----------------------------------> // Declaring multiple variables
<!-- ----------------------------------> let currentTemp = 22, favNumber, favColor = "Red";
<!-- ---------------------------------->
<!-- ----------------------------------> // Declaring a constant. A variable whose value can't be changed after being initialized
<!-- ----------------------------------> const ROOM_TEMP_C = 21.5 (conventionally for identifier to be upper case and have underscores)
<!-- ---------------------------------->
<!-- ----------------------------------> || Literals ||
<!-- ----------------------------------> 
<!-- ----------------------------------> // Numeric Literals
<!-- ----------------------------------> 22, 22.5, 
<!-- ----------------------------------> 
<!-- ----------------------------------> // String Literals
<!-- ----------------------------------> 'G', "Goat"
<!-- ---------------------------------->
<!-- ----------------------------------> // Passing values between variables
<!-- ----------------------------------> let num1 = 2, num2 = 3;
<!-- ----------------------------------> num1 = num2;
<!-- ---------------------------------->
<!-- ----------------------------------> || Primitives &amp; Objects
<!-- ----------------------------------> 
<!-- ----------------------------------> //Primitives are immutable where they are intrinsively unchanging.  
<!-- ---------------------------------->
<!-- ----------------------------------> // List of primitives include:  
<!-- ----------------------------------> Numbers, Strings, Booleans, Null, Undefined, and Symbols
<!-- ---------------------------------->
<!-- ----------------------------------> // Objects unlike primitives can take on different forms and values
<!-- ---------------------------------->
<!-- ----------------------------------> // Objects include:
<!-- ----------------------------------> Array, Date, RegExp, Map and WeakMap, Set and WeakSet, Number, String, and Boolean
<!-- ---------------------------------->
<!-- ----------------------------------> || Numbers ||
<!-- ---------------------------------->
<!-- ----------------------------------> // There are four types of numeric literals that include:
<!-- ----------------------------------> -integer literal : 10 
<!-- ----------------------------------> -hexadecimal literal: 0x0000ff
<!-- ----------------------------------> -octal literal: 0o0022
<!-- ----------------------------------> -decimal literal: 21.5
<!-- ----------------------------------> -exponential literal: 3.0e6 (3.0 x 10^6)
<!-- ----------------------------------> -infinity: Infinity
<!-- ----------------------------------> - negative infinity: -Infinity
<!-- ----------------------------------> -Not a Number: NaN
<!-- ---------------------------------->
<!-- ----------------------------------> //Other numeric values provided by the Number object
<!-- ----------------------------------> const small = Number.EPSILON;
<!-- ----------------------------------> const bigInt = Number.MAX_SAFE_INTEGER;
<!-- ----------------------------------> const max = Number.MAX_VALUE;
<!-- ----------------------------------> const minInt = Number.MIN_SAFE_INTEGER;
<!-- ---------------------------------->
<!-- ----------------------------------> || Strings ||
<!-- ---------------------------------->
<!-- ----------------------------------> // Strings are represented by Unicode Characters
<!-- ---------------------------------->
<!-- ----------------------------------> //Escaping
<!-- ----------------------------------> const dialog1 = "He looked up and said \"don't do that!\" to Max.";
<!-- ----------------------------------> const dialog2 = 'He looked up and said "don\'t do that!" to Max.';
<!-- ----------------------------------> const s = "In JavaScript, use \\ as an escape character in strings.";
<!-- ---------------------------------->
<!-- ----------------------------------> // String Concatenation
<!-- ----------------------------------> let currentTemp = 19.5;
<!-- ----------------------------------> const message = "The current temperature is " + currentTemp + "\u00b0C"; (Note: \u00b0C is a unicode point for the "degree" symbol)
<!-- ---------------------------------->
<!-- ----------------------------------> // String Interpolation
<!-- ----------------------------------> let currentTemp = 19.5;
<!-- ----------------------------------> const message = `The current temperature is ${currentTemp}\u00b0C`;
<!-- ---------------------------------->
<!-- ----------------------------------> || Number, String, and Boolean Objects ||
<!-- ---------------------------------->
<!-- ----------------------------------> // These objects serve to just store special values (e.g. Number.EPSILON), and to provide functionalikty in the form of a function as such:
<!-- ----------------------------------> const s = "example";
<!-- ----------------------------------> s.toUpperCase(); // "example"
<!-- ---------------------------------->
<!-- ----------------------------------> //As shown in the case above, despite the fact thats s stores a string literal which is a primitive type, we are able to call String object function due to Javascript automatically assigning a temporary String object to that variable. This is proven using the following demonstration where we try to initialize a new property for the object:
<!-- ---------------------------------->
<!-- ----------------------------------> s.rating = 3;
<!-- ----------------------------------> s.rating; // undefined

<!-- ----------------------------------> || Arrays ||
<!-- ---------------------------------->
<!-- ----------------------------------> //Initailizing arrays
<!-- ----------------------------------> const a1 = [1, 2, 3, 4]; 
<!-- ----------------------------------> const a2 = [1, "two", 3, null]; // array with different types
<!-- ----------------------------------> const a3 = ["What the hammer",    //array on multiple lines
<!-- ---------------------------------->             "What the crab apples",
<!-- ---------------------------------->             "Check in on the don"
<!-- ---------------------------------->            ];
<!-- ----------------------------------> const a4 = [    //array containing objects
<!-- ---------------------------------->    {name: "Ruby", hardness: 9},
<!-- ---------------------------------->    {name: "Diamond", hardness: 10},
<!-- ---------------------------------->    {name: "Topaz", hardness: 8}
<!-- ----------------------------------> ];
<!-- ----------------------------------> const a5 = [ //array containing arrays
<!-- ---------------------------------->     [1,2,3],
<!-- ---------------------------------->     ["one", "two", "three"],
<!-- ---------------------------------->     [true, false, true]
<!-- ----------------------------------> ];
<!-- ----------------------------------> 
<!-- ----------------------------------> //Array length property
<!-- ----------------------------------> a1.length; // 4
<!-- ---------------------------------->
<!-- ----------------------------------> //Retrieving Array Data using numeric index
<!-- ----------------------------------> a1[0]; // 1 (first item in the array)
<!-- ----------------------------------> a1[a1.length - 1]; // 4 (last item in the array)
<!-- ---------------------------------->
<!-- ----------------------------------> //Overwriting items in an array
<!-- ----------------------------------> a1[0] = 5; // a1 is now [5,2,3,4]
<!-- ---------------------------------->
<!-- ----------------------------------> || Date Objects ||
<!-- ----------------------------------> 
<!-- ----------------------------------> //Initializing a date object with the current date and time
<!-- ----------------------------------> const now = new Date();
<!-- ----------------------------------> now; // example: Thu Aug 20 2015 18:31:26 GMT-0700 (Pacific Daylight Time)
<!-- ----------------------------------> 
<!-- ----------------------------------> // Initializing a date object to a specified date
<!-- ----------------------------------> const halloween = new Date(2016, 9, 31);
<!-- ----------------------------------> 
<!-- ----------------------------------> // Initialize a date object to specified date AND Time
<!-- ----------------------------------> const halloweenParty = new Date(2016, 9, 31, 19, 0); // 19:00 = 7:00pm
<!-- ---------------------------------->
<!-- ----------------------------------> // Retrieving components from a date
<!-- ----------------------------------> halloweenParty.getFullYear(); // 2016
<!-- ----------------------------------> halloweenParty.getMonth();    // 9
<!-- ----------------------------------> halloweenParty.getDate();     // 31

<!-- ----------------------------------> || Regular Expression Objects ||
<!-- ---------------------------------->
<!-- ----------------------------------> //Regular expressions are represented as RegExp objects and have a literal syntax consisting of symbols between a pair of forward slashes
<!-- ---------------------------------->
<!-- ----------------------------------> // examples
<!-- ----------------------------------> const emailFilter = /\b[a-z0-9._-]+@[a-z_-]+(?:\.[a-z]+)+\b/;
<!-- ----------------------------------> const phoneNumberFilter = /(:?\+1)?(:?\(\d{3}\)\s?|\d{3}[\s-]?)\d{3}[\s-]?\d{4}/;

<!-- ----------------------------------> || Data Type Conversations ||
<!-- ---------------------------------->
<!-- ----------------------------------> // Converting to numbers using Number constructor
<!-- ----------------------------------> const numStr = "33.3";
<!-- ----------------------------------> const num = Number(numStr); // Passes a number value (NOT a Number object)
<!-- ----------------------------------> 
<!-- ----------------------------------> //Converting to numbers using Number functions
<!-- ----------------------------------> const a = parseInt("16 volts", 10);  // "volts" ignored and 16 is parsed in base 10
<!-- ----------------------------------> const b = parseInt("3a", 16); // "3a" is parsed in hexdecimal to 58 
<!-- ----------------------------------> comnst c = parseFloat("15.5 kph"); // "kph" is ignored; parseFloat always assumes base 10
<!-- ----------------------------------> 
<!-- ----------------------------------> //Dates converted to numbers
<!-- ----------------------------------> const d = new Date();
<!-- ----------------------------------> const ts = d.valueOf(); //miliseconds since midnight of Jan 1, 1970 UTC

<!-- ----------------------------------> 
<!-- ----------------------------------> //All objects have toString() methods that return string representations for those objects
<!-- ---------------------------------->
<!-- ----------------------------------> const n = 33.5;
<!-- ----------------------------------> const s = n.toString(); // string "33.5" assigned to s
<!-- ----------------------------------> 
<!-- ----------------------------------> //Array toString()
<!-- ----------------------------------> const arr = [1, true, "hello"];
<!-- ----------------------------------> arr.toString(); // "1, true, hello"
